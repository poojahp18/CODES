#include <stdio.h>


int missingNum(int arr[],int n)
{
    int sum1=n*(n+1)/2,sum2=0;
    int i;
    for(i=0;i<n-1;i++)
    {
        sum2+=arr[i];
    }
    
    return sum1-sum2;
}

int main()
{
    int test,num;
    int arr[1000];
    int missing=0;
    int i=0,j=0;
    
    scanf("%d",&test);
    for(i=0;i<test;i++)
    {
        scanf("%d",&num);
        for(j=0;j<num-1;j++)
        {
            scanf("%d",&arr[j]);
        }
        
        missing=missingNum(arr,num);
        printf("%d\n",missing);
    }
}
