#include <stdio.h>

int main() {
	//code
	
	int arr[1000];
	int test=0,num=0;
	int i=0,j=0;
	int max=0;
	int val=-1;
	
	scanf("%d",&test);
	for(i=0;i<test;i++)
	{
	    scanf("%d",&num);
	    
	    scanf("%d",&arr[0]);
	    max=arr[0];
	    
	    for(j=1;j<num;j++)
	    {
	        scanf("%d",&arr[i]);
	        printf("%d ",(max<=arr[i])?val:arr[i]);
	        max=arr[i];
	        
	    }
	    printf("%d\n",val);
	}
	
	return 0;
}
